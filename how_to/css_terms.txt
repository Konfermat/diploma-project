/* Базовый сброс стилей */
  /* * - все элементы */
  /* *::before - псевдоэлементы до (виртуальные элементы) */
  /* *::after - псевдоэлементы после (виртуальные элементы) */

  *,
  *::before,
  *::after {
    box-sizing: border-box;
  }

    outline-offset: 0;
  }

/* Для адаптивности размеров шрифта */

    font: 62.5% / 1.15 consolas;
  }
