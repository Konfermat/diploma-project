        setlocal enabledelayedexpansion

        :: 1. Получаем ширину экрана из команды "mode con"
        for /f "tokens=2 delims=:" %%A in ('mode con ^| findstr Columns') do (
            set /a width=%%A
        )

        :: Убираем возможные лишние пробелы из переменной
        set /a width=%width%

        :: 2. Генерируем строку из символов # нужной длины
        set "line="
        for /l %%I in (1, 1, %width%) do (
            set "line=!line!#"
        )

        :: 3. Выводим строку на экран
        echo !line!

    "Курсы и тесты"
        Цель сервиса
            Редак

        Описание сервиса
            Сервис имеет две основные идейные функции:
                Основаная:
                    Функция конструктора курсов.
                        Создание, удаление, редактирование, хранение и прохождение курсов.
                        Преподаватель создает курс, его разбивает на уроки с текстовым материалом и тестами к ним.
                        
                Второстепенная(эксперементальная):
                    Ученик сам придумывает тесты по курсу и отвечает по нему

            
        Возможности сервиса.            
            Создание, удаление, редактирование, прохождение курсов.
            Курсами можно будет поделиться с помощью ссылки или qr кода.
